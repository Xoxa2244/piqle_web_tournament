# 1. DATABASE_URL (Transaction mode - рекомендуется для Vercel)
# Получить: Settings → Database → Connection string → Transaction mode
# Формат: postgresql://postgres.xxxxx:ПАРОЛЬ@aws-1-us-east-2.pooler.supabase.com:6543/postgres?pgbouncer=true&prepared_statements=false
DATABASE_URL=postgresql://postgres.xxxxx:ВАШ_ПАРОЛЬ@aws-1-us-east-2.pooler.supabase.com:6543/postgres?pgbouncer=true&prepared_statements=false

# Альтернатива: Session mode
# DATABASE_URL=postgresql://postgres.xxxxx:ВАШ_ПАРОЛЬ@aws-1-us-east-2.pooler.supabase.com:5432/postgres

# 2. NEXT_PUBLIC_SUPABASE_URL
# Получить: Settings → API → Project URL
NEXT_PUBLIC_SUPABASE_URL=https://xxxxx.supabase.co

# 3. NEXT_PUBLIC_SUPABASE_ANON_KEY
# Получить: Settings → API → Project API keys → anon public
NEXT_PUBLIC_SUPABASE_ANON_KEY=ваш_новый_anon_ключ_здесь

# 4. SUPABASE_SERVICE_ROLE_KEY
# Получить: Settings → API → Project API keys → service_role
# ⚠️ ВНИМАНИЕ: НЕ ПОКАЗЫВАЙТЕ ЭТОТ КЛЮЧ НИКОМУ!
SUPABASE_SERVICE_ROLE_KEY=ваш_новый_service_role_ключ_здесь

# ИНСТРУКЦИЯ:

# 4. Убедитесь, что выбраны все окружения (Production, Preview, Development)
# 5. Сохраните изменения
